```python
import sys
input = sys.stdin.readline

N= int(input())
TimeList = list(map(int,input().split()))

TimeList = sorted(TimeList)
sumv = 0
for i in range(1,N+1):
    sumv += sum([TimeList[j] for j in range(i)])

print(sumv)
```
